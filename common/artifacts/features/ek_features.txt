	group = animal_hide_type
	trigger = {
		OR = {
			scope:owner.var:animal_type = flag:senche_cougar
			AND = {
				scope:owner.var:animal_type = flag:none
				scope:location ?= {
					OR = {
						geographical_region = mundus_tamriel_elsweyr

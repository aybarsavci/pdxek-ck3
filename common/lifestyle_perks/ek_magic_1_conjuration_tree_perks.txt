conjuration_novice_perk = {
	lifestyle = magic_lifestyle
	tree = conjuration
	position = { 0 0 }
	icon = node_diplomacy
	
	auto_selection_weight = {
		value = 11
		if = {
			limit = { has_students_magicka_background = yes }
			add = 1989
		}
		if = {
			limit = {
				exists = dynasty
				dynasty = { has_dynasty_modifier = magicka_legacy_conjuration_focus }
			}
			add = 500
		}
		if = {
            limit = { culture = { has_cultural_parameter = cheaper_conjuration_spells } } ###EK EDIT: new parameter
            add = 200
        }
		if = {
			limit = { religion = { is_in_family = rf_daedra } }
			add = 200
		}
		if = {
			limit = {
				religion = { is_in_family = rf_aedra }
				NOT = { has_trait = cynical }
			}
			add = -200
		}
		if = {
			limit = {
				can_start_new_lifestyle_tree_trigger = no
			}
			multiply = 0
		}
		if = {
			limit = {
				can_magic_trigger = no
			}
			multiply = 0
		}
	}
	
	effect = {	
		custom_description_no_bullet = {
			text = conjuration_novice_perk_effect
		}
		custom_description_no_bullet = {
			text = conjuration_novice_perk_effect_necromancy
		}
		if = {
            limit = { culture = { has_cultural_parameter = cheaper_conjuration_spells } } ###EK EDIT: new parameter
            add_magic_lifestyle_xp = 330
        }
	}
}

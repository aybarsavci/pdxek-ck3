illusion_novice_perk = {
	lifestyle = magic_lifestyle
	tree = illusion
	position = { 0 1 }
	icon = node_intrigue
	
	auto_selection_weight = {
		value = 11
		if = {
			limit = { has_students_magicka_background = yes }
			add = 1989
		}
		if = {
            limit = { culture = { has_cultural_parameter = cheaper_illusion_spells } } ###EK EDIT: new parameter
            add = 200
        }
		if = {
            limit = { has_trait = paranoid }
            add = 20
        }
		if = {
            limit = { has_trait = deceitful }
            add = 20
        }
		if = {
			limit = {
				can_start_new_lifestyle_tree_trigger = no
			}
			multiply = 0
		}
		if = {
			limit = {
				can_magic_trigger = no
			}
			multiply = 0
		}
	}
	
	parent = illusion_apprentice_perk
	
	effect = {	
		custom_description_no_bullet = {
			text = illusion_novice_perk_effect
		}
		if = {
            limit = { culture = { has_cultural_parameter = cheaper_illusion_spells } } ###EK EDIT: new parameter
            add_magic_lifestyle_xp = 330
        }
	}
}

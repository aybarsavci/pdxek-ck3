restoration_novice_perk = {
	lifestyle = magic_lifestyle
	tree = restoration
	position = { 0 0 }
	icon = node_learning
	
	auto_selection_weight = {
		value = 11
		if = {
			limit = { has_students_magicka_background = yes }
			add = 1989
		}
		if = {
			limit = {
				exists = dynasty
				dynasty = { has_dynasty_modifier = magicka_legacy_restoration_focus }
			}
			add = 500
		}
		if = {
            limit = { culture = { has_cultural_parameter = cheaper_restoration_spells } } ###EK EDIT: new parameter
            add = 200
        }
		if = {
			limit = {
				has_trait = zealous
				religion = { is_in_family = rf_aedra }
			}
			add = 200
		}
		if = {
			limit = { 
				religion = { is_in_family = rf_daedra } 
				NOR = {
					religion = religion:meridia_religion
					religion = religion:velothi_religion
				}
			}
			add = -200
		}
		if = {
			limit = { has_trait = compassionate }
			add = 20
		}
		if = {
			limit = {
				can_start_new_lifestyle_tree_trigger = no
			}
			multiply = 0
		}
		if = {
			limit = {
				is_ai = yes
			}
			multiply = 0
		}
	}
	
	effect = {	
		custom_description_no_bullet = {
			text = restoration_novice_perk_effect
		}
		#custom_description_no_bullet = {
		#	text = enable_ek_life_extension_scheme
		#}
        if = {
            limit = { culture = { has_cultural_parameter = cheaper_restoration_spells } } ###EK EDIT: new parameter
            add_magic_lifestyle_xp = 330
        }
    }
}

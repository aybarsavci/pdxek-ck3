dark_arts_mysticism_focus = {
	lifestyle = dark_arts_lifestyle
	
	modifier = {
		intrigue = 2
		learning = 2
		courtier_and_guest_opinion = -5
	}

	desc = {
		desc = dark_arts_mysticism_focus_desc
		desc = line_break
	}

	is_valid = {
		is_necromancer = yes
	}

	auto_selection_weight = {
		value = 0
		if = {
			limit = { 
				is_necromancer = yes
				NOT = { has_perk = necromancy_master_perk }
			}
			add = 2000
		}
	}
}

### Arcana focus
magic_theory_focus = { 
	lifestyle = magic_lifestyle
	
	modifier = {
		learning = 3
		stewardship = 1
	}

	desc = {
		desc = magic_theory_focus_desc
		desc = line_break
	}

	is_valid = {
		can_magic_trigger = yes
	}

	auto_selection_weight = {
		value = 100
		if = {
			limit = { 
				OR = { 
					has_students_magicka_background = yes
					is_court_mage_trigger = yes
				}
			}
			add = 1900
		}
		if = { 
			limit = { culture = { has_cultural_parameter = magic_preference } }
			add = 100
		}
	}
}

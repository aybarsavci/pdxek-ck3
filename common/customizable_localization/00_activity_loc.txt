GetActivitySpecialType = { # Only used for Pilgrimages
	type = activity

	#Pilgrimage
	##Pious Type
	text = {
		trigger = {
			has_activity_type = activity_pilgrimage
			OR = {
				has_activity_option = {
					category = special_type
					option = pilgrimage_type_pious
				}
				#EK DISABLED: no hajj
				#has_activity_option = {
				#	category = special_type
				#	option = pilgrimage_type_hajj
				#}
			}
			
		}
		localization_key = pilgrimage_piousness
	}
	
	##Worldly Type
	text = {
		trigger = {
			has_activity_type = activity_pilgrimage
			has_activity_option = {
				category = special_type
				option = pilgrimage_type_worldly
			}
		}
		localization_key = pilgrimage_worldliness
	}
	
	##Uni Visit
	#EK EDIT: added normal education option
	text = {
		trigger = {
			has_activity_type = activity_adult_education
			has_activity_option = {
				category = special_type
				option = adult_education_option_mundane
			}			
		}
		localization_key = activity_adult_education_success
	}
	#EK EDIT END

	##Camp Party
	text = {
		trigger = { has_activity_type = activity_camp_party }
		localization_key = activity_camp_party_success
	}
	
	#EK EDIT: add our own feast types
	text = {
		trigger = {
			has_activity_type = activity_feast
			has_activity_option = {
				category = special_type
				option = feast_type_generic
			}
		}
		localization_key = activity_feast
	}


	text = {
		trigger = {
			has_activity_type = activity_feast

				category = special_type
				option = feast_type_new_life
			}
		}

GetCharacterContestStatus = {
	type = character

	text = {
		trigger = {
			is_in_guest_subset = {
				name = spectator
			}
		}
		localization_key = locale_window_status_spectating
	}

	text = {
		trigger = {
			involved_activity = {
				is_current_phase_active = yes
			}
			is_in_guest_subset = {
				name = qualified
			}
			exists = var:progress_to_victory
		}
		localization_key = locale_window_status_participating
	}

	text = {
		trigger = {
			involved_activity = {
				is_current_phase_active = yes
			}
			is_in_guest_subset = {
				name = contestant
			}
			not = {
				is_in_guest_subset = {
					name = qualified
				}
			}
		}
		localization_key = locale_window_status_not_qualify
	}

	text = {
		trigger = {
			involved_activity = {
				is_current_phase_active = yes
			}
			is_in_guest_subset = {
				name = qualified
			}
			not = { exists = var:progress_to_victory }
		}
		localization_key = locale_window_status_knocked_out
	}

	text = {
		trigger = {
			involved_activity = {
				is_current_phase_active = yes
			}
			exists = var:progress_to_victory
		}
		localization_key = locale_window_status_participating
	}
}

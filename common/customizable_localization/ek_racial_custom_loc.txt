HandClawPaw = { #Singular
	type = character
	
	text = {
		trigger = { race_has_claws_trigger = yes }
		localization_key = bodypart_claw
	}

	text = {
		trigger = { race_has_paws_trigger = yes }
		localization_key = bodypart_paw
	}

	text = {
		localization_key = bodypart_hand
		fallback = yes
	}
}

SkinTypeIsAre = { #Grammer
	type = character
	
	text = {
		trigger = {
			OR = {
				race_has_scales_trigger = yes

			}
		}
		localization_key = verb_are
	}

		localization_key = verb_is

GetPatronStatus = {
	type = character

	text = {
		trigger = {
			exists = var:patron_shown
			switch = {
				trigger = var:patron_shown
				flag:azura = { global_var:status_azura = flag:ascendant }
				flag:boethiah = { global_var:status_boethiah = flag:ascendant }
				flag:clavicus = { global_var:status_clavicus = flag:ascendant }
				flag:hermaeus = { global_var:status_hermaeus = flag:ascendant }
				flag:hircine = { global_var:status_hircine = flag:ascendant }
				flag:malacath = { global_var:status_malacath = flag:ascendant }
				flag:mehrunes = { global_var:status_mehrunes = flag:ascendant }
				flag:mephala = { global_var:status_mephala = flag:ascendant }
				flag:meridia = { global_var:status_meridia = flag:ascendant }
				flag:molag = { global_var:status_molag = flag:ascendant }
				flag:namira = { global_var:status_namira = flag:ascendant }
				flag:nocturnal = { global_var:status_nocturnal = flag:ascendant }
				flag:peryite = { global_var:status_peryite = flag:ascendant }
				flag:sanguine = { global_var:status_sanguine = flag:ascendant }
				flag:sheogorath = { global_var:status_sheogorath = flag:ascendant }
				flag:vaermina = { global_var:status_vaermina = flag:ascendant }
			}
		}
		localization_key = patron_status_ascendant
	}

	text = {
		trigger = {
			exists = var:patron_shown
			switch = {
				trigger = var:patron_shown
				flag:azura = { global_var:status_azura = flag:convalescing }
				flag:boethiah = { global_var:status_boethiah = flag:convalescing }
				flag:clavicus = { global_var:status_clavicus = flag:convalescing }
				flag:hermaeus = { global_var:status_hermaeus = flag:convalescing }
				flag:hircine = { global_var:status_hircine = flag:convalescing }
				flag:malacath = { global_var:status_malacath = flag:convalescing }
				flag:mehrunes = { global_var:status_mehrunes = flag:convalescing }
				flag:mephala = { global_var:status_mephala = flag:convalescing }
				flag:meridia = { global_var:status_meridia = flag:convalescing }
				flag:molag = { global_var:status_molag = flag:convalescing }
				flag:namira = { global_var:status_namira = flag:convalescing }
				flag:nocturnal = { global_var:status_nocturnal = flag:convalescing }
				flag:peryite = { global_var:status_peryite = flag:convalescing }
				flag:sanguine = { global_var:status_sanguine = flag:convalescing }
				flag:sheogorath = { global_var:status_sheogorath = flag:convalescing }
				flag:vaermina = { global_var:status_vaermina = flag:convalescing }
			}
		}
		localization_key = patron_status_convalescing
	}

	text = {
		trigger = {
			exists = var:patron_shown
			switch = {
				trigger = var:patron_shown
				flag:azura = { global_var:status_azura = flag:cataclysm }
				flag:boethiah = { global_var:status_boethiah = flag:cataclysm }
				flag:clavicus = { global_var:status_clavicus = flag:cataclysm }
				flag:hermaeus = { global_var:status_hermaeus = flag:cataclysm }
				flag:hircine = { global_var:status_hircine = flag:cataclysm }
				flag:malacath = { global_var:status_malacath = flag:cataclysm }
				flag:mehrunes = { global_var:status_mehrunes = flag:cataclysm }
				flag:mephala = { global_var:status_mephala = flag:cataclysm }
				flag:meridia = { global_var:status_meridia = flag:cataclysm }
				flag:molag = { global_var:status_molag = flag:cataclysm }
				flag:namira = { global_var:status_namira = flag:cataclysm }
				flag:nocturnal = { global_var:status_nocturnal = flag:cataclysm }
				flag:peryite = { global_var:status_peryite = flag:cataclysm }
				flag:sanguine = { global_var:status_sanguine = flag:cataclysm }
				flag:sheogorath = { global_var:status_sheogorath = flag:cataclysm }
				flag:vaermina = { global_var:status_vaermina = flag:cataclysm }
			}
		}
		localization_key = patron_status_cataclysm
	}

	text = {
		trigger = { always = no }
		fallback = yes
		localization_key = patron_status_nominal
	}
}

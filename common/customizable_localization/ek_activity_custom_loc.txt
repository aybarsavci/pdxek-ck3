GetHistBlessingTrait = {
	type = character

	#Amber Hist
	text = {
		trigger = { 
			faith = { has_doctrine = tenet_amberhist }
			NOT = { 
				has_trait = amber_blessed			
			}
		}
		localization_key = activity_hist_communion_amber_blessed_1_desc
	}

	text = {
		trigger = { 
			faith = { has_doctrine = tenet_amberhist }
			has_trait = amber_blessed_1
		}
		localization_key = activity_hist_communion_amber_blessed_2_desc
	}

	text = {
		trigger = { 
			faith = { has_doctrine = tenet_amberhist }
			has_trait = amber_blessed_2
		}
		localization_key = activity_hist_communion_amber_blessed_3_desc
	}

	text = {
		trigger = { 
			faith = { has_doctrine = tenet_amberhist }
			has_trait = amber_blessed_3
		}
		localization_key = activity_hist_communion_amber_blessed_3_desc
	}

	#Sun Hist
		text = {
		trigger = { 
			faith = { has_doctrine = tenet_sunhist }
			NOT = { 
				has_trait = sun_blessed			
			}
		}
		localization_key = activity_hist_communion_sun_blessed_1_desc
	}

	text = {
		trigger = { 
			faith = { has_doctrine = tenet_sunhist }
			has_trait = sun_blessed_1
		}
		localization_key = activity_hist_communion_sun_blessed_2_desc
	}

	text = {
		trigger = { 
			faith = { has_doctrine = tenet_sunhist }
			has_trait = sun_blessed_2
		}
		localization_key = activity_hist_communion_sun_blessed_3_desc
	}

	text = {
		trigger = { 
			faith = { has_doctrine = tenet_sunhist }
			has_trait = sun_blessed_3
		}
		localization_key = activity_hist_communion_sun_blessed_3_desc
	}

	#Ghost Hist
		text = {
		trigger = { 
			faith = { has_doctrine = tenet_ghosthist }
			NOT = { 
				has_trait = ghost_scale				
			}
		}
		localization_key = activity_hist_communion_ghost_scale_1_desc
	}

	text = {
		trigger = { 
			faith = { has_doctrine = tenet_ghosthist }
			has_trait = ghost_scale_1
		}
		localization_key = activity_hist_communion_ghost_scale_2_desc
	}

	text = {
		trigger = { 
			faith = { has_doctrine = tenet_ghosthist }
			has_trait = ghost_scale_2
		}
		localization_key = activity_hist_communion_ghost_scale_3_desc
	}

	text = {
		trigger = { 
			faith = { has_doctrine = tenet_ghosthist }
			has_trait = ghost_scale_3
		}
		localization_key = activity_hist_communion_ghost_scale_3_desc
	}

	#Fallback if none of the above
	text = {
		trigger = { 
			NOR = {
				faith = { has_doctrine = tenet_amberhist }
				faith = { has_doctrine = tenet_sunhist }
				faith = { has_doctrine = tenet_ghosthist }
			}
		}
		localization_key = activity_hist_communion_standard_blessing_desc
	}

}

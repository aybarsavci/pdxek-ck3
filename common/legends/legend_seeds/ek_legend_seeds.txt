olaf_one_eye = { #Legend for Whiterun
	type = legitimizing
	quality = famed
	is_shown = {
		has_dlc_feature = legends
		NOT = {
			has_global_variable = olaf_one_eye_legend
		}
		OR = {
			AND = { #Nord holding Whiterun
				culture = { has_cultural_pillar = heritage_atmoran }
				has_title = title:k_whiterun
			}
			dynasty = dynasty:310006 #Yngvlar - dynasty of Whiterun
		}
	}
	is_valid = {
		custom_tooltip = {
			text = must_whiterun_tt
			OR = {
				has_title = title:k_whiterun
			}
		}
		prestige_level >= very_high_prestige_level
	}

	chronicle = ancestral_glory
	chronicle_properties = {
		ancestor = character:numinex_100
		title = title:c_whiterun.empire
	}
}

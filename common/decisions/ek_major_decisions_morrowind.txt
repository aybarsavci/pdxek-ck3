accept_scarab_blood_decision = { #Accept Scarab blood, based on witch decision
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
	}
	decision_group_type = major
	ai_check_interval = 120
	desc = accept_scarab_blood_decision_desc

	is_shown = {
		exists = dynasty
		dynasty = { has_dynasty_modifier = unmourned_blood_dynasty_modifier }
		NOT = { dynasty = { has_dynasty_modifier = scarab_blood_dynasty_modifier } }
	}
	is_valid = {
		is_dynast = yes
		OR = {
			faith = faith:sixth_house
			custom_tooltip = {
				text = has_secret_sixth_house_faith
				secret_faith ?= faith:sixth_house
			}
		}
		trigger_if = {
			limit = {
				exists = player_heir
				player_heir = {
					is_child_of = root
					dynasty = root.dynasty
				}
			}
			player_heir = {
				is_sixth_house_known_by_trigger = { CHARACTER = root }
			}
		}
		#We want a certain % to be witches but also a minimum amount. So if the house is small, we use a minimum number, and if the house is larger we use %
		trigger_if = {
			limit = {
				dynasty = {
					any_dynasty_member = {
						is_adult = yes
						count <= 6
					}
				}
			}
			dynasty = {
				custom_tooltip = {
					text = accept_scarab_blood_house_member_count_tt
					any_dynasty_member = {
						is_adult = yes
						count >= accept_scarab_blood_member_count_value
						is_sixth_house_known_by_trigger = { CHARACTER = root }
					}
				}
				any_dynasty_member = {
					is_adult = yes
					count >= accept_scarab_blood_member_count_value
					is_sixth_house_known_by_trigger = { CHARACTER = root }
				}
			}
		}
		trigger_else = {
			dynasty = {
				custom_tooltip = {
					text = accept_scarab_blood_house_member_percent_tt
					any_dynasty_member = {
						is_adult = yes
						percent >= accept_scarab_blood_member_percent_value
						is_sixth_house_known_by_trigger = { CHARACTER = root }
					}
				}
				any_dynasty_member = {
					is_adult = yes
					percent >= accept_scarab_blood_member_percent_value
					is_sixth_house_known_by_trigger = { CHARACTER = root }
				}
			}
		}
	}

	is_valid_showing_failures_only = {
	}

	effect = {
		accept_scarab_blood_decision_effects = yes
	}
	ai_will_do = { #Do it always, for coherence.
		base = 100
	}
}

	picture = { reference = "gfx/interface/illustrations/decisions/decision_misc.dds" }
	decision_group_type = major
	
	desc = buy_writ_of_execution_desc
	selection_tooltip = buy_writ_of_execution_tooltip

	cooldown = { years = 2 }
	
	is_shown = {
		any_character_situation = {
			any_participant_group = {
				participant_group_type = grand_council_members
				participant_group_has_character = root
			}
		}
		any_scheme = {
			scheme_type = murder			
			NOT = { has_scheme_modifier = writ_of_execution_modifier }
			NOT = { has_scheme_modifier = morag_tong_hired }
		}
	}

	is_valid = {
		custom_description = {
			text = buy_writ_of_execution_trigger
			any_scheme = {
				scheme_type = murder
			}
		}
		custom_description = {
			text = buy_writ_of_execution_outlander_trigger
			trigger_if = {
				limit = {
					any_scheme = {
						scheme_type = murder
						scheme_target_character = {
							save_temporary_scope_as = target
							NOT = {
								any_character_situation = {
									any_participant_group = {
										participant_group_type = grand_council_members
										participant_group_has_character = scope:target
									}
								}
							}
						}
					}
				}
				any_character_situation = {
					any_situation_sub_region = {
						has_sub_region_phase_parameter = can_buy_writ_of_execution_against_anyone
						any_situation_sub_region_participant_group = {
							participant_group_type = grand_council_members
							participant_group_has_character = root
						}
					}
				}
			}
		}
	}
		
	effect = {
		custom_tooltip = buy_writ_of_execution_custom_desc
		if = {
			limit = {
				any_scheme = {
					scheme_type = murder
					count = 1
				}
			}
			random_scheme = {
				limit = {
					scheme_type = murder
				}
				hidden_effect = {
					add_scheme_modifier = {
						type = writ_of_execution_modifier
					}
				}
			}
			trigger_event = ek_major_decisions.0040
		}
		if = {
			limit = {

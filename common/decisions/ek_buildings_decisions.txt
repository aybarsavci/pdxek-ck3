## Destroy Bridges of Leyawiin
destroy_bridges_leyawiin_decision = {
	picture = { reference ="gfx/interface/illustrations/decisions/decision_port.dds" }
	
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { NOT = { has_title = title:c_leyawiin } }
				desc = destroy_bridges_leyawiin_decision_desc_vassal
			}
			desc = destroy_bridges_leyawiin_decision_desc_direct
		}
	}
	
	selection_tooltip = destroy_bridges_leyawiin_decision_tooltip
	decision_group_type = ek_sb

	ai_check_interval = 120

	is_shown = {
		# It's in your realm and the bridges can't have already been disabled
		completely_controls = title:c_leyawiin
		
		NOT = { capital_county = title:c_leyawiin }
		
		province:2525 = { has_building = bridges_leyawiin_02 }
	}

	is_valid = {
		trigger_if = {
			limit = { NOT = { has_title = title:c_leyawiin } } # A vassal holds it
			
			title:c_leyawiin.holder = {
				OR = {
					opinion = {
						target = ROOT
						value >= 75
					}
					root = { has_strong_hook = title:c_leyawiin.holder }
				}
			}
		}
		
		trigger_if = {
			limit = { has_title = title:c_leyawiin } # Hold it yourself

			custom_tooltip = {
				text = destroy_bridges_leyawiin_decision_control.tt
				title:c_leyawiin.county_control >= 75
			}
		}
	}

	is_valid_showing_failures_only = {
		is_alive = yes
		NOT = { exists = involved_activity }
		is_imprisoned = no
	}

	effect = {
		custom_tooltip = destroy_bridges_leyawiin_decision_enables_buildings_imperial_isle_ct
		custom_tooltip = destroy_bridges_leyawiin_decision_enables_port_buildings_rumare_ct
		
		province:2525 = {
			remove_building = bridges_leyawiin_02
			add_special_building = bridges_leyawiin_01
        }

		if = {
			limit = { NOT = { has_title = title:c_leyawiin } } # A vassal holds it
			
			title:c_leyawiin.holder = {
				add_opinion = {
					target = root
					modifier = disappointed_opinion
					opinion = -25
				}
			}
		}

		if = {
			limit = { has_title = title:c_leyawiin } # Hold it yourself

			title:c_leyawiin = {
				change_county_control = -25
			}
		}
	}

	cost = {
		prestige = 1000
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}

form_crown_republic_decision = { #Decision to form e_crown
	picture = { reference ="gfx/interface/illustrations/decisions/hegathe.dds" }

	desc = form_crown_republic_decision_desc
	selection_tooltip = form_crown_republic_decision_tooltip
	decision_group_type = major

	ai_check_interval = 120

	is_shown = {
		is_ruler = yes
		is_landed = yes
		NOT = { exists = title:e_crown.holder } #Crown doesn't exist
		#Has Crown tradition
		culture = { has_cultural_tradition = tradition_crown_core }
		#Only kings
		highest_held_title_tier = tier_kingdom
	}

	is_valid = {
		is_independent_ruler = yes
		prestige_level >= 3
		piety_level >= 3 
		calc_true_if = {
			amount >= 2
			completely_controls_region = ek_region_crownlands
			completely_controls_region = ek_region_northern_crownlands
			completely_controls = title:d_sentinel
			completely_controls = title:d_taneth
			completely_controls = title:d_rihad
			completely_controls = title:d_ojwambu
        }
	}

	is_valid_showing_failures_only = {
		is_capable_adult = yes
		is_imprisoned = no
		is_independent_ruler = yes
	}

	cost = {
		prestige = 500
		gold = 500
	}

	effect = {
		form_crown_republic_scripted_effect = yes
		#No global flag, can happen several times
	}
	
	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}

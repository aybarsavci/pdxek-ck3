story_cycle_pet_skeever = {

	on_setup = {
		# Has the gender already been specified?
		if = { 
			limit = {
				story_owner = { exists = var:skeever_gender }
			}
			if = {
				limit = {
					story_owner = { var:skeever_gender = flag:male }
				}
				assign_skeever_gender_effect = { GENDER = male }
			}
			else_if = {
				limit = {
					story_owner = { var:skeever_gender = flag:female }
				}
				assign_skeever_gender_effect = { GENDER = female }
			}
		}
		else = {
			assign_skeever_gender_effect = { GENDER = random }
		}
		# Has the fur-color already been specified?
		if = { 
			limit = {
				story_owner = { exists = var:skeever_color }
			}
			if = {
				limit = {
					story_owner = { var:skeever_color = flag:grey }
				}
				assign_skeever_color_effect = { COLOR = grey }
			}
			if = {
				limit = {
					story_owner = { var:skeever_color = flag:white }
				}
				assign_skeever_color_effect = { COLOR = white }
			}
			if = {
				limit = {
					story_owner = { var:skeever_color = flag:black }
				}
				assign_skeever_color_effect = { COLOR = black }
			}
		}
		else = {
			assign_skeever_color_effect = { COLOR = random }
		}
		
		### Setup the rest
		set_variable = {
			name = skeever_age_variable
			value = 0
		}

		story_owner = {
			add_character_modifier = {
				modifier = skeever_story_modifier
			}
			add_character_flag = had_skeever_story
		}

	}

	on_end = {
		story_owner = {
			remove_skeever_story_modifiers_effect = yes
			remove_skeever_name_effect = yes
		}
	}

	on_owner_death = {
		scope:story = { end_story = yes }
	}

	# Give the skeever a name!
	effect_group = {
		days = { 2 4 }
		chance = 100
	
		triggered_effect = {
			trigger = {
				story_owner = {
					NOT = { has_character_flag = is_naming_skeever }
				}
				NOT = { exists = story_owner.var:story_cycle_skeever_name }
			}
			effect = {
				story_owner = {
					add_character_flag = is_naming_skeever
					trigger_event = ek_pet_animal.6001
				}
			}
		}
	}

	#To save the name in the story's scope (can for some reason not be done in the naming event itself)
	effect_group = {
		days = 5
		chance = 100
	
		triggered_effect = {
			trigger = {
				exists = story_owner.var:story_cycle_skeever_name
			}
			effect = {
				set_variable = {
					name = story_cycle_skeever_name
					value = scope:story.story_owner.var:story_cycle_skeever_name
				}
			}
		}
	}

	#Age the skeever
	effect_group = {
		days = 365

		trigger = {
			exists = var:skeever_age_variable
		}

		triggered_effect = {
			trigger = { always = yes }
			effect = {
				change_variable = {
					name = skeever_age_variable
					add = 1
				}
			}
		}
	}

	# The skeever dies :(
	effect_group = {
		days = { 1095 1461 }
		chance = 100
	
		triggered_effect = {
			trigger = {
				exists = story_owner.var:story_cycle_skeever_name
			}
			effect = {
				story_owner = {
					add_character_flag = {
						flag = skeever_is_dying
					}
					trigger_event = ek_pet_animal.6199
				}
			}
		}
	}

	# Something happens - random events
	effect_group = {
		days = { 365 600 }
		# days = { 7 10 }
		chance = 50

		trigger = {
			exists = story_owner.var:story_cycle_skeever_name
			story_owner = {
				NOT = {
					has_character_flag = skeever_is_dying
				}
			}
		}

		first_valid = {
			triggered_effect = {
				trigger = { always = yes }
				effect = {
					story_owner = {
						trigger_event = {
							on_action = ongoing_skeever_events
						}
					}
				}
			}
		}
	}
}

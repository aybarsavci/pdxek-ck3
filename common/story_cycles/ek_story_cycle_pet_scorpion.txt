story_cycle_pet_scorpion = {

	on_setup = {
		# Has the gender already been specified?
		if = { 
			limit = {
				story_owner = { exists = var:scorpion_gender }
			}
			if = {
				limit = {
					story_owner = { var:scorpion_gender = flag:male }
				}
				assign_scorpion_gender_effect = { GENDER = male }
			}
			else_if = {
				limit = {
					story_owner = { var:scorpion_gender = flag:female }
				}
				assign_scorpion_gender_effect = { GENDER = female }
			}
		}
		else = {
			assign_scorpion_gender_effect = { GENDER = random }
		}
		# Has the color already been specified?
		if = { 
			limit = {
				story_owner = { exists = var:scorpion_color }
			}
			if = {
				limit = {
					story_owner = { var:scorpion_color = flag:coffee }
				}
				assign_scorpion_color_effect = { COLOR = coffee }
			}
			else_if = {
				limit = {
					story_owner = { var:scorpion_color = flag:red }
				}
				assign_scorpion_color_effect = { COLOR = red }
			}
			else_if = {
				limit = {
					story_owner = { var:scorpion_color = flag:cerulean }
				}
				assign_scorpion_color_effect = { COLOR = cerulean }
			}
			else_if = {
				limit = {
					story_owner = { var:scorpion_color = flag:brown }
				}
				assign_scorpion_color_effect = { COLOR = brown }
			}
		}
		else = {
			assign_scorpion_color_effect = { COLOR = random }
		}
		
		### Setup the rest
		set_variable = {
			name = scorpion_age_variable
			value = 0
		}

		story_owner = {
			add_character_modifier = {
				modifier = scorpion_story_modifier
			}
			add_character_flag = had_scorpion_story
		}

	}

	on_end = {
		story_owner = {
			remove_scorpion_story_modifiers_effect = yes
			remove_scorpion_name_effect = yes
		}
	}

	on_owner_death = {
		scope:story = { end_story = yes }
	}

	# Give the scorpion a name!
	effect_group = {
		days = { 2 4 }
		chance = 100
	
		triggered_effect = {
			trigger = {
				story_owner = {
					NOT = { has_character_flag = is_naming_scorpion }
				}
				NOT = { exists = story_owner.var:story_cycle_scorpion_name }
			}
			effect = {
				story_owner = {
					add_character_flag = is_naming_scorpion
					trigger_event = ek_pet_animal.1001
				}
			}
		}
	}

	#To save the name in the story's scope (can for some reason not be done in the naming event itself)
	effect_group = {
		days = 5
		chance = 100
	
		triggered_effect = {
			trigger = {
				exists = story_owner.var:story_cycle_scorpion_name
			}
			effect = {
				set_variable = {
					name = story_cycle_scorpion_name
					value = scope:story.story_owner.var:story_cycle_scorpion_name
				}
			}
		}
	}

	#Age the scorpion
	effect_group = {
		days = 365

		trigger = {
			exists = var:scorpion_age_variable
		}

		triggered_effect = {
			trigger = { always = yes }
			effect = {
				change_variable = {
					name = scorpion_age_variable
					add = 1
				}
			}
		}
	}

	# The scorpion dies :(
	effect_group = {
		days = { 7000 10000 }
		chance = 100
	
		triggered_effect = {
			trigger = {
				exists = story_owner.var:story_cycle_scorpion_name
			}
			effect = {
				story_owner = {
					add_character_flag = {
						flag = scorpion_is_dying
					}
					trigger_event = ek_pet_animal.1199
				}
			}
		}
	}

	# Something happens - random events
	effect_group = {
		days = { 365 600 }
		# days = { 7 10 }
		chance = 50

		trigger = {
			exists = story_owner.var:story_cycle_scorpion_name
			story_owner = {
				NOT = {
					has_character_flag = scorpion_is_dying
				}
			}
		}

		first_valid = {
			triggered_effect = {
				trigger = { always = yes }
				effect = {
					story_owner = {
						trigger_event = {
							on_action = ongoing_scorpion_events
						}
					}
				}
			}
		}
	}
}

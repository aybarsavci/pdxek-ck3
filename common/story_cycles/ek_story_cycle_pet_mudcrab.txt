story_cycle_pet_mudcrab = {

	on_setup = {
		# Has the gender already been specified?
		if = { 
			limit = {
				story_owner = { exists = var:mudcrab_gender }
			}
			if = {
				limit = {
					story_owner = { var:mudcrab_gender = flag:male }
				}
				assign_mudcrab_gender_effect = { GENDER = male }
			}
			else_if = {
				limit = {
					story_owner = { var:mudcrab_gender = flag:female }
				}
				assign_mudcrab_gender_effect = { GENDER = female }
			}
		}
		else = {
			assign_mudcrab_gender_effect = { GENDER = random }
		}
		# Has the fur-color already been specified?
		if = { 
			limit = {
				story_owner = { exists = var:mudcrab_color }
			}
			if = {
				limit = {
					story_owner = { var:mudcrab_color = flag:blue }
				}
				assign_mudcrab_color_effect = { COLOR = blue }
			}
			if = {
				limit = {
					story_owner = { var:mudcrab_color = flag:coral }
				}
				assign_mudcrab_color_effect = { COLOR = coral }
			}
			if = {
				limit = {
					story_owner = { var:mudcrab_color = flag:brown }
				}
				assign_mudcrab_color_effect = { COLOR = brown }
			}
		}
		else = {
			assign_mudcrab_color_effect = { COLOR = random }
		}
		
		### Setup the rest
		set_variable = {
			name = mudcrab_age_variable
			value = 0
		}

		story_owner = {
			add_character_modifier = {
				modifier = mudcrab_story_modifier
			}
			add_character_flag = had_mudcrab_story
		}

	}

	on_end = {
		story_owner = {
			remove_mudcrab_story_modifiers_effect = yes
			remove_mudcrab_name_effect = yes
		}
	}

	on_owner_death = {
		scope:story = { end_story = yes }
	}

	# Give the mudcrab a name!
	effect_group = {
		days = { 2 4 }
		chance = 100
	
		triggered_effect = {
			trigger = {
				story_owner = {
					NOT = { has_character_flag = is_naming_mudcrab }
				}
				NOT = { exists = story_owner.var:story_cycle_mudcrab_name }
			}
			effect = {
				story_owner = {
					add_character_flag = is_naming_mudcrab
					trigger_event = ek_pet_animal.5001
				}
			}
		}
	}

	#To save the name in the story's scope (can for some reason not be done in the naming event itself)
	effect_group = {
		days = 5
		chance = 100
	
		triggered_effect = {
			trigger = {
				exists = story_owner.var:story_cycle_mudcrab_name
			}
			effect = {
				set_variable = {
					name = story_cycle_mudcrab_name
					value = scope:story.story_owner.var:story_cycle_mudcrab_name
				}
			}
		}
	}

	#Age the mudcrab
	effect_group = {
		days = 365

		trigger = {
			exists = var:mudcrab_age_variable
		}

		triggered_effect = {
			trigger = { always = yes }
			effect = {
				change_variable = {
					name = mudcrab_age_variable
					add = 1
				}
			}
		}
	}

	# The mudcrab dies :(
	effect_group = {
		days = { 1095 1461 }
		chance = 100
	
		triggered_effect = {
			trigger = {
				exists = story_owner.var:story_cycle_mudcrab_name
			}
			effect = {
				story_owner = {
					add_character_flag = {
						flag = mudcrab_is_dying
					}
					trigger_event = ek_pet_animal.5199
				}
			}
		}
	}

	# Something happens - random events
	effect_group = {
		days = { 365 600 }
		# days = { 7 10 }
		chance = 50

		trigger = {
			exists = story_owner.var:story_cycle_mudcrab_name
			story_owner = {
				NOT = {
					has_character_flag = mudcrab_is_dying
				}
			}
		}

		first_valid = {
			triggered_effect = {
				trigger = { always = yes }
				effect = {
					story_owner = {
						trigger_event = {
							on_action = ongoing_mudcrab_events
						}
					}
				}
			}
		}
	}
}

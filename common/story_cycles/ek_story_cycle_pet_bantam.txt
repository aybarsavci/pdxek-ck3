story_cycle_pet_bantam = {

	on_setup = {
		# Has the gender already been specified?
		if = { 
			limit = {
				story_owner = { exists = var:bantam_gender }
			}
			if = {
				limit = {
					story_owner = { var:bantam_gender = flag:male }
				}
				assign_bantam_gender_effect = { GENDER = male }
			}
			else_if = {
				limit = {
					story_owner = { var:bantam_gender = flag:female }
				}
				assign_bantam_gender_effect = { GENDER = female }
			}
		}
		else = {
			assign_bantam_gender_effect = { GENDER = random }
		}
		# Has the fur-color already been specified?
		if = { 
			limit = {
				story_owner = { exists = var:bantam_color }
			}
			if = {
				limit = {
					story_owner = { var:bantam_color = flag:tan }
				}
				assign_bantam_color_effect = { COLOR = tan }
			}
		}
		else = {
			assign_bantam_color_effect = { COLOR = random }
		}
		
		### Setup the rest
		set_variable = {
			name = bantam_age_variable
			value = 0
		}

		story_owner = {
			add_character_modifier = {
				modifier = bantam_story_modifier
			}
			add_character_flag = had_bantam_story
		}

	}

	on_end = {
		story_owner = {
			remove_bantam_story_modifiers_effect = yes
			remove_bantam_name_effect = yes
		}
	}

	on_owner_death = {
		scope:story = { end_story = yes }
	}

	# Give the bantam a name!
	effect_group = {
		days = { 2 4 }
		chance = 100
	
		triggered_effect = {
			trigger = {
				story_owner = {
					NOT = { has_character_flag = is_naming_bantam }
				}
				NOT = { exists = story_owner.var:story_cycle_bantam_name }
			}
			effect = {
				story_owner = {
					add_character_flag = is_naming_bantam
					trigger_event = ek_pet_animal.3001
				}
			}
		}
	}

	#To save the name in the story's scope (can for some reason not be done in the naming event itself)
	effect_group = {
		days = 5
		chance = 100
	
		triggered_effect = {
			trigger = {
				exists = story_owner.var:story_cycle_bantam_name
			}
			effect = {
				set_variable = {
					name = story_cycle_bantam_name
					value = scope:story.story_owner.var:story_cycle_bantam_name
				}
			}
		}
	}

	#Age the bantam
	effect_group = {
		days = 365

		trigger = {
			exists = var:bantam_age_variable
		}

		triggered_effect = {
			trigger = { always = yes }
			effect = {
				change_variable = {
					name = bantam_age_variable
					add = 1
				}
			}
		}
	}

	# The bantam dies :(
	effect_group = {
		days = { 1825 3650 }
		chance = 100
	
		triggered_effect = {
			trigger = {
				exists = story_owner.var:story_cycle_bantam_name
			}
			effect = {
				story_owner = {
					add_character_flag = {
						flag = bantam_is_dying
					}
					trigger_event = ek_pet_animal.3199
				}
			}
		}
	}

	# Something happens - random events
	effect_group = {
		days = { 365 600 }
		# days = { 7 10 }
		chance = 50

		trigger = {
			exists = story_owner.var:story_cycle_bantam_name
			story_owner = {
				NOT = {
					has_character_flag = bantam_is_dying
				}
			}
		}

		first_valid = {
			triggered_effect = {
				trigger = { always = yes }
				effect = {
					story_owner = {
						trigger_event = {
							on_action = ongoing_bantam_events
						}
					}
				}
			}
		}
	}
}

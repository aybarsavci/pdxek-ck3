story_cycle_pet_vvardvark = {

	on_setup = {
		# Has the gender already been specified?
		if = { 
			limit = {
				story_owner = { exists = var:vvardvark_gender }
			}
			if = {
				limit = {
					story_owner = { var:vvardvark_gender = flag:male }
				}
				assign_vvardvark_gender_effect = { GENDER = male }
			}
			else_if = {
				limit = {
					story_owner = { var:vvardvark_gender = flag:female }
				}
				assign_vvardvark_gender_effect = { GENDER = female }
			}
		}
		else = {
			assign_vvardvark_gender_effect = { GENDER = random }
		}
		# Has the fur-color already been specified?
		if = { 
			limit = {
				story_owner = { exists = var:vvardvark_color }
			}
			if = {
				limit = {
					story_owner = { var:vvardvark_color = flag:tan }
				}
				assign_vvardvark_color_effect = { COLOR = tan }
			}
			if = {
				limit = {
					story_owner = { var:vvardvark_color = flag:tuxedo }
				}
				assign_vvardvark_color_effect = { COLOR = tuxedo }
			}
			if = {
				limit = {
					story_owner = { var:vvardvark_color = flag:dozen_banded }
				}
				assign_vvardvark_color_effect = { COLOR = dozen_banded }
			}
		}
		else = {
			assign_vvardvark_color_effect = { COLOR = random }
		}
		
		### Setup the rest
		set_variable = {
			name = vvardvark_age_variable
			value = 0
		}

		story_owner = {
			add_character_modifier = {
				modifier = vvardvark_story_modifier
			}
			add_character_flag = had_vvardvark_story
		}

	}

	on_end = {
		story_owner = {
			remove_vvardvark_story_modifiers_effect = yes
			remove_vvardvark_name_effect = yes
		}
	}

	on_owner_death = {
		scope:story = { end_story = yes }
	}

	# Give the vvardvark a name!
	effect_group = {
		days = { 2 4 }
		chance = 100
	
		triggered_effect = {
			trigger = {
				story_owner = {
					NOT = { has_character_flag = is_naming_vvardvark }
				}
				NOT = { exists = story_owner.var:story_cycle_vvardvark_name }
			}
			effect = {
				story_owner = {
					add_character_flag = is_naming_vvardvark
					trigger_event = ek_pet_animal.4001
				}
			}
		}
	}

	#To save the name in the story's scope (can for some reason not be done in the naming event itself)
	effect_group = {
		days = 5
		chance = 100
	
		triggered_effect = {
			trigger = {
				exists = story_owner.var:story_cycle_vvardvark_name
			}
			effect = {
				set_variable = {
					name = story_cycle_vvardvark_name
					value = scope:story.story_owner.var:story_cycle_vvardvark_name
				}
			}
		}
	}

	#Age the vvardvark
	effect_group = {
		days = 365

		trigger = {
			exists = var:vvardvark_age_variable
		}

		triggered_effect = {
			trigger = { always = yes }
			effect = {
				change_variable = {
					name = vvardvark_age_variable
					add = 1
				}
			}
		}
	}

	# The vvardvark dies :(
	effect_group = {
		days = { 8000 8766 }
		chance = 100
	
		triggered_effect = {
			trigger = {
				exists = story_owner.var:story_cycle_vvardvark_name
			}
			effect = {
				story_owner = {
					add_character_flag = {
						flag = vvardvark_is_dying
					}
					trigger_event = ek_pet_animal.4199
				}
			}
		}
	}

	# Something happens - random events
	effect_group = {
		days = { 365 600 }
		# days = { 7 10 }
		chance = 50

		trigger = {
			exists = story_owner.var:story_cycle_vvardvark_name
			story_owner = {
				NOT = {
					has_character_flag = vvardvark_is_dying
				}
			}
		}

		first_valid = {
			triggered_effect = {
				trigger = { always = yes }
				effect = {
					story_owner = {
						trigger_event = {
							on_action = ongoing_vvardvark_events
						}
					}
				}
			}
		}
	}
}

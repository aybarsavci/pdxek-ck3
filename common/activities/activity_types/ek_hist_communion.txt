activity_hist_communion = {
	is_shown = {
		highest_held_title_tier > tier_barony
		is_landed = yes        
        has_hist_communion = yes
		trigger_if = {
			limit = {
				is_ai = yes
			}
			years_from_game_start >= 3
			is_at_war = no
			short_term_gold >= hist_communion_base_cost
		}
	}

	can_start_showing_failures_only = {	
		is_available_adult = yes
		custom_tooltip = {
			text = hist_communion_needs_hist
			any_sub_realm_county = {
				any_county_province = {
					has_building_or_higher = hist_02
				}
			}
		}
	}

	is_valid = {
		scope:host = {
			#If you don't have a capital to return to, don't bother
			exists = capital_province
			is_capable_adult = yes
			is_imprisoned = no
			is_landed = yes 
			NOT = { is_incapable = yes }
			#Invalidate flag, until/unless this is fixed in code
			NOT = { has_variable = pilgrimage_invalidated_illness }
		}
		# Converting mid-pilgrimage is a big no-no.
		scope:activity = {
			OR = {
				NOT = { has_variable = starting_faith }
				# We could check whether the place you're going to is still a holy site in the new faith, but then that messes up all the people moving with you, so we just cancel the whole thing.
				scope:host.faith = var:starting_faith
			}
		}
	}

	can_always_plan = no

	on_invalidated = {
		scope:host = {
			#We're heading back home to bury the old host
			if = {
				limit = { has_variable = pilgrimage_death_turn_back }
				trigger_event = pilgrimage.0042
			}			
		}
		# Host becomes unlanded
		if = {
			limit = {
				scope:host = { is_landed = no }
			}
			scope:activity = {
				activity_type = { save_scope_as = activity_type }
			}
			every_attending_character = {
				trigger_event = activity_system.0320
			}
		}
		if = {
			limit = {
				scope:host = { is_incapable = yes }
			}
			scope:activity = {
				activity_type = { save_scope_as = activity_type }
			}
			scope:host = {
				trigger_event = activity_system.0330
			}
			every_attending_character = {
				limit = { NOT = { this = scope:host } }
				trigger_event = activity_system.0331
			}
		}
	}

	on_host_death = {
		#Make the player heir the new host, if they're on the Pilgrimage
		if = {
			limit = { 
				exists = scope:host.player_heir
				any_attending_character = { this = scope:host.player_heir }
				scope:host.faith = scope:host.player_heir.faith
			}
			scope:host = { save_scope_as = old_host }
			scope:host.player_heir = { save_scope_as = new_host }
			#Also inherit the travel plan
			if = {
				limit = {
					scope:host.current_travel_plan = {
						any_entourage_character = { this = scope:new_host }
					}
				}
				scope:host.current_travel_plan = { set_travel_plan_owner = scope:new_host }
			}
			set_activity_host = scope:new_host
			scope:new_host = {
				set_variable = {
					name = old_host_to_be_buried
					value = scope:old_host #The old host
				}
				add_character_flag = pilgrimage_dealing_with_invalidation
				trigger_event = activity_system.0081
			}
		}
		#Otherwise indirectly cancel the activity by not setting a new host
		else = {
			# Send your entourage home
			scope:host = {
				current_travel_plan = {
					every_entourage_character = { set_location_to_default = yes }
				}
			}
		}
	}

	is_location_valid = {
		custom_tooltip = {
			text = activity_hist_communion_no_hist_tree
			has_building_or_higher = hist_02
		}
		custom_tooltip = { 
			text = activity_hist_communion_hist_tree_sapling
			NOT = { has_building = hist_01 }
		}
		custom_tooltip = { 
			text = activity_hist_communion_hist_tree_faith
			county.holder.faith = root.faith
		}	
	}

	province_description = {
		desc = activity_hist_communion_province_desc
		#Is it a young Hist?
		triggered_desc = {
			trigger = {
				has_building = hist_02
			}
			desc = hist_communion_young_hist_desc
		}
		#Is it a mature Hist?
		triggered_desc = {
			trigger = {
				has_building = hist_03
			}
			desc = hist_communion_mature_hist_desc
		}
		#Is it an old Hist?
		triggered_desc = {
			trigger = {
				has_building = hist_04
			}
			desc = hist_communion_old_hist_desc
		}
		#Is it an ancient Hist?
		triggered_desc = {
			trigger = {
				has_building = hist_05
			}
			desc = hist_communion_ancient_hist_desc
		}
	}

	province_score = {
		value = 0
		# Young Hist
		if = {
			limit = {
				county = {
					any_county_province = {
						has_building = hist_02
					}
				}
			}
			add = 10
		}
		# Mature Hist
		if = {
			limit = {
				county = {
					any_county_province = {
						has_building = hist_03
					}
				}
			}
			add = 25
		}
		# Old Hist
		if = {
			limit = {
				county = {
					any_county_province = {
						has_building = hist_04
					}
				}
			}
			add = 50
		}
		# Ancient Hist
		if = {
			limit = {
				county = {
					any_county_province = {
						has_building = hist_05
					}
				}
			}
			add = 100
		}
	}

	ai_will_do = {
		value = 30
		if = {
			limit = {
				ai_energy < 0
			}

				add = ai_energy
				desc = "ai_energy"
			}

			limit = { short_term_gold >= major_gold_value }

			}

			}
		}
		if = {
			limit = {
				ai_zeal > 0
				piety < 0
			}
			add = {
				value = 10
				desc = "Zealous wants above 0 piety"
			}
		}
		if = {
			limit = {
				is_independent_ruler = no
				has_vassal_stance = zealot
			}
			add = {
				value = 10
				desc = zealot
			}
		}
		if = {
			limit = {
				OR = {
					has_trait = cynical
					any_secret = { secret_type = secret_non_believer }
				}
			}
			add = {
				value = -30
				desc = "Is cynical or non-believer"
			}
		}
		if = {
			limit = {
				OR = {
					has_trait = ghost_scale_3
					has_trait = amber_blessed_3
					has_trait = sun_blessed_3
				}
			}
			add = {
				value = -30
				desc = "Has maxed out blessing"
			}
		}
		if = {
			limit = {
				can_be_healed_by_hist_trigger = yes
			}
			add = {
				value = 30
				desc = "Seeks healing"
			}
		}
		# Legitimacy
		if = {
			limit = {
				is_valid_for_legitimacy_change = yes
			}
			if = { # Low Legitimacy
				limit = {
					OR = {
						legitimacy_level = 1
						legitimacy_level = 2
					}
				}
				add = {
					value = 10
					desc = "low_legitimacy"
				}
			}
			if = { # Very Low Legitimacy
				limit = {
					legitimacy_level = 0
				}
				add = {
					value = 20
					desc = "low_legitimacy"
				}
			}
		}
	}

		if = {

		}

		scope:activity ?= {
			activity_setup_special_type_progression_effect = yes

			# Check what faith the communion is supposed to be for.
			set_variable = {
				name = starting_faith
				value = scope:host.faith
			}

			# How jazzed are people about going?
			#EK TODO: this is not working. If Paradox fixes the same thing with pilgrimages, apply the fix here

				limit = {
					NOT = {
						this = scope:host
					}
				}
				# Then, apply suitable opinion.
				custom = custom.every_entourage_character
				if = {
					limit = {
						NOT = {
							has_opinion_modifier = {
								modifier = brought_me_on_a_hist_communion_opinion
								target = scope:host
							}
						}
					}
					# Ancient Hist
					if = {
						limit = { scope:activity.var:hist_age = flag:ancient_hist }
						add_opinion = {
							target = scope:host
							modifier = brought_me_on_a_hist_communion_opinion
							opinion = 20
						}
					}
					# Old Hist
					if = {
						limit = { scope:activity.var:hist_age = flag:old_hist }
						add_opinion = {
							target = scope:host
							modifier = brought_me_on_a_hist_communion_opinion
							opinion = 15
						}
					}
					# Mature Hist
					if = {
						limit = { scope:activity.var:hist_age = flag:mature_hist }
						add_opinion = {
							target = scope:host
							modifier = brought_me_on_a_hist_communion_opinion
							opinion = 10
						}
					}
					# Young Hist
					if = {
						limit = { scope:activity.var:hist_age = flag:young_hist }
						add_opinion = {
							target = scope:host
							modifier = brought_me_on_a_hist_communion_opinion
							opinion = 5
						}
					}
				}
			}
		}

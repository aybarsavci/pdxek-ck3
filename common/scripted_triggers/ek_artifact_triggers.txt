# portrait wield triggers
portrait_should_wield_staff_trigger = {
	trigger_if = {
		limit = { any_equipped_character_artifact = { artifact_slot_type = primary_armament } }
		any_equipped_character_artifact = { artifact_type = staff }
	}
	trigger_else = {
		has_variable = signature_weapon
		var:signature_weapon = flag:staff
	}
}

		}
	}
}

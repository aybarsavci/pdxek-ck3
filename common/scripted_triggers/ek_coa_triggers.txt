coa_altmer_title_trigger = {	#For baronies / mercenaries / and leased temples
	AND = {
		coa_altmer_culture_trigger = yes
		OR = {
			highest_held_title_tier = tier_barony
			primary_title ?= { is_mercenary_company = yes }
			government_has_flag = government_is_theocracy
			AND = {
				exists = liege
				liege = {
					any_sub_realm_barony = {
						lessee = ROOT
						title_province = {
							has_holding_type = church_holding
						}
					}
				}
			}
		}
	}
}

coa_hist_title_trigger = {	#For hist holdings
	OR = {
		AND = {
			coa_hist_religion_trigger = yes
			government_has_flag = government_is_theocracy
		}
		AND = {
			root.faith = { has_doctrine_parameter = can_hold_hist_holdings }
			root.capital_barony.title_province ?= { has_holding_type = hist_holding }
		}
	}
}

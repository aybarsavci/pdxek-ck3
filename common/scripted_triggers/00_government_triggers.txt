should_apply_tribal_development_penalty_trigger = {
	government_has_flag = government_is_tribal
	NOR = {
		any_liege_or_above = {
			has_feudal_like_government = yes
		}
		any_vassal_or_below = {
			has_feudal_like_government = yes
		}
		any_neighboring_top_liege_realm_owner = {
			has_feudal_like_government = yes
			tier_difference = {
				target = prev
				value < -1
			}
		}
		scope:county ?= {
			holder ?= {
				any_neighboring_realm_same_rank_owner = {
					has_feudal_like_government = yes
				}
			}
		}
	}
}

court_mage_court_position = {
	sort_order = 395 # right after court physician
	max_available_positions = 1

	skill = learning

	court_position_asset = { #Nords get Clever Man
		trigger = {
			root.culture ?= {
				has_cultural_tradition = tradition_nord_core
			}
		}
		animation = holding_staff
		background = "gfx/interface/illustrations/event_scenes/fp1_throneroom_tribal.dds"
		localization_key = court_mage_clever_man
	}

	court_position_asset = { #Landless get Field Mage
		trigger = {
			government_has_flag = government_is_landless_adventurer
			# landless nords can keep their Clever Man
			NOT = {
				root.culture ?= {
					has_cultural_tradition = tradition_nord_core
				}
			}
		}
		animation = holding_staff
		background = "gfx/interface/illustrations/event_scenes/ep3_campfire.dds"
		localization_key = court_mage_field_mage
	}

	court_position_asset = {
		animation = holding_staff
		background = "gfx/interface/illustrations/event_scenes/throneroom_west.dds"
	}

	opinion = {
		value = regular_court_position_opinion
	}

	# How high the aptitude has to be to reach the 5 different modifier levels
	aptitude_level_breakpoints = { 20 40 60 80 }

	# How good is this character in this position? root scope is the holder of the court position
	aptitude = {
		value = 1
		add = court_position_aptitude_boon_voyager_trait
		add = {
			value = arcana
			multiply = 1.5
			max = 50
			desc = court_position_skill_arcana
		}
		add = aptitude_from_arcana_traits
		add = aptitude_from_alteration_perks
		add = aptitude_from_conjuration_perks
		add = aptitude_from_destruction_perks 
		add = aptitude_from_illusion_perks
		add = aptitude_from_restoration_perks
		add = aptitude_from_necromancy_perks

		add = court_position_aptitude_traveling_friends_value
		add = court_position_aptitude_family_business_value
		add = court_position_aptitude_low_penalty_value
		# Seneschal
		if = {
			limit = {
				liege.court_position:seneschal_court_position ?= {
					has_variable = performing_improve_court
				}
			}
			add = {
				value = liege.court_position:seneschal_court_position.seneschal_improve_court_value
				desc = seneschal_improve_court_aptitude_desc
			}
		}
	}

	is_shown = {}

	is_shown_character = {
		scope:employee = {
			OR = {
				is_courtier_of = scope:liege
				is_vassal_of = scope:liege
			}
			is_mindless_character = no
		}
	}

	valid_position = {
		always = yes
	}

	valid_character = {
		valid_camp_officer_trigger = {
			OFFICER = court_mage
		}
	}

	revoke_cost = {
		prestige =  {
			value = major_court_position_prestige_revoke_cost
			# Set to 0 via obligation hook
			if = {
				limit = {
					court_position_obligation_hook_salary_trigger = {
						POS = court_mage
						LIEGE = root
					}
				}
				multiply = 0
			}
			# Avoid AI prestige issues
			if = {
				limit = {
					exists = this
					is_ai = yes
				}
				multiply = 0
			}
			# Special case for events
			if = {
				limit = {
					exists = this
					has_character_flag = free_revoke_cost
				}
				multiply = 0
			}
			#Lowering for LAAMPS, to match Camp Officer revoke cost
			if = {
				limit = { 
					exists = this
					government_has_flag = government_is_landless_adventurer
				}
				value = minor_court_position_prestige_revoke_cost
			}
		}
	}

	salary = {
		round = no
		gold = {
			value = medium_court_position_salary # Update X_salary_tooltip if script_value is changed
			# Set to 0 with Dynasty Perk Inner Circle
			if = {
				limit = { court_position_inner_circle_salary_trigger = { POS = court_mage } }

					value = 0

					court_position_obligation_hook_salary_trigger = {
						POS = court_mage
						LIEGE = scope:liege
					}
				}
				multiply = {
					value = 0.5
					desc = obligation_hook_salary
					format = "BASE_VALUE_FORMAT_DECIMALS_MINUS_POSITIVE"
				}
			}
			# Halved salary
			if = {
				limit = {
					court_position_halved_salary_trigger = { POS = court_mage }
				}
				multiply = {
					value = 0.5
					desc = court_position_halved_salary
					format = "BASE_VALUE_FORMAT_DECIMALS_MINUS_POSITIVE"
				}
			}
			# Increased salary
			if = {
				limit = {
					court_position_increased_salary_trigger = { POS = court_mage }
				}
				multiply = {
					value = 1.5
					desc = court_position_increased_salary
					format = "BASE_VALUE_FORMAT_DECIMALS_MINUS_POSITIVE"
				}
			}
			#You're working with LAAMPS now
			if = {
				limit = { 
					scope:liege ?= { government_has_flag = government_is_landless_adventurer }
				}
				multiply = {
					value = 0
					desc = court_position_laamps_free
					format = "BASE_VALUE_FORMAT_DECIMALS_MINUS_POSITIVE"
				}
			}
		}
		prestige = {
			value = medium_court_position_prestige_salary # Update X_salary_tooltip if script_value is changed
			# Set to 0 with Dynasty Perk Inner Circle
			if = {
				limit = { court_position_inner_circle_salary_trigger = { POS = court_mage } }
				multiply = {
					value = 0
					desc = inner_circle_salary_mod
					format = "BASE_VALUE_FORMAT_DECIMALS_MINUS_POSITIVE"
				}
			}
			# Halved via obligation hook
			else_if = {
				limit = {
					court_position_obligation_hook_salary_trigger = {
						POS = court_mage
						LIEGE = scope:liege
					}
				}
				multiply = {
					value = 0.5
					desc = obligation_hook_salary
					format = "BASE_VALUE_FORMAT_DECIMALS_MINUS_POSITIVE"
				}
			}
			# Halved salary
			if = {
				limit = {
					court_position_halved_salary_trigger = { POS = court_mage }
				}
				multiply = {
					value = 0.5
					desc = court_position_halved_salary
					format = "BASE_VALUE_FORMAT_DECIMALS_MINUS_POSITIVE"
				}
			}
			# Increased salary
			if = {
				limit = {
					court_position_increased_salary_trigger = { POS = court_mage }
				}
				multiply = {
					value = 1.5
					desc = court_position_increased_salary
					format = "BASE_VALUE_FORMAT_DECIMALS_MINUS_POSITIVE"
				}
			}
			#You're working with LAAMPS now
			if = {
				limit = { 
					scope:liege ?= { government_has_flag = government_is_landless_adventurer }
				}
				multiply = {
					value = 0
					desc = court_position_laamps_free
					format = "BASE_VALUE_FORMAT_DECIMALS_MINUS_POSITIVE"
				}
			}
		}
	}

	is_powerful_agent = yes

	# These modifiers are employed based on what level the aptitude reaches (levels are defined as breakpoints) 
	custom_scaling_employer_modifier_description = {
		terrible = court_mage_terrible_desc
		poor = court_mage_poor_desc
		average = court_mage_average_desc
		good = court_mage_good_desc
		excellent = court_mage_excellent_desc
		range = court_mage_range_desc
	}

	custom_employer_modifier_description = court_mage_employer_custom_effect_description

	modifier = {
		learning = 1
		#arcana = 2 #fake skill, effect applied in ek_magic_values.txt
	}

	on_court_position_received = {
		basic_gained_court_position_effect = yes
	}
	on_court_position_revoked = {
		basic_revoked_court_position_effect = yes
	}
	on_court_position_invalidated = {
		basic_invalidated_court_position_effect = yes
	}
	on_court_position_vacated = {
		court_position_vacated_event_effect = {
			COURT_POSITION_TYPE = court_mage_court_position
		}
	}

	candidate_score = {
		value = 150
        add = scope:base_value
        add = court_position_candidate_score_base_value
        add = court_position_candidate_aptitude_value

			limit = { exists = scope:employee.var:magicka }
			add = {
				value = scope:employee.var:magicka
				desc = game_concept_magicka
			}
		}
		# AI is less likely to fire the court mage, unless they have a better candidate.
        if = {
            limit = {
                exists = scope:firing_court_position
            }
            add = 200
            scope:employee = {
                # Does the liege have a better candidate that they should be hiring instead?
                if = {
                    limit = {
                        scope:highest_available_aptitude > scope:employee_aptitude
                    }
                    add = -150
                }
            }
        }

		if = { # To stop petitioned positions immediately being fired by the AI
			limit = {
				court_position_petition_ai_weight_trigger = { ROLE = court_mage }
			}
			add = 1000
		}
	}

	search_for_courtier = {
		send_interface_toast = {
			title = hire_court_mage_decision
			custom_tooltip = hire_court_mage_decision_effect_tooltip
		}
		trigger_event = {
			id = ek_court_position.0001
			days = 14
		}
	}
}

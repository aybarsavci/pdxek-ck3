debug_change_grand_council = {
	category = interaction_debug_mpo
	use_diplomatic_range = no
	ignores_pending_interaction_block = yes
	common_interaction = yes
	icon = travel_mixed

	interface_priority = -1
	auto_accept = yes

	is_shown = {
		debug_only = yes
		scope:actor = {
			any_character_situation = {
				this = situation:grand_council
			}
		}
	}
	send_option = {
		flag = autonomy
		localization = debug_change_gcm_autonomy
	}
	send_option = {
		flag = centralisation
		localization = debug_change_gcm_centralisation
	}
	send_option = {
		flag = balanced
		localization = debug_change_gcm_balanced
	}
	on_accept = {
		switch = {
			trigger = yes
			scope:autonomy = {
				if = {
					limit = {
						NOT = { exists = situation:grand_council }
					}
					start_situation = {
						type = grand_council
						start_phase = gcm_autonomy
					}
				}
				else = {
					situation:grand_council = {
						every_situation_sub_region = {
							change_phase = { phase = gcm_autonomy }
						}
					}
				}
			}
			scope:centralisation = {
				if = {
					limit = {
						NOT = { exists = situation:grand_council }
					}
					start_situation = {
						type = grand_council
						start_phase = gcm_centralised
					}
				}
				else = {
					situation:grand_council = {
						every_situation_sub_region = {
							change_phase = { phase = gcm_centralised }
						}
					}
				}
			}
			scope:balanced = {
				if = {
					limit = {
						NOT = { exists = situation:grand_council }
					}
					start_situation = {
						type = grand_council
						start_phase = gcm_balanced
					}
				}
				else = {
					situation:grand_council = {
						every_situation_sub_region = {
							change_phase = { phase = gcm_balanced }
						}
					}
				}
			}
		}
	}
}

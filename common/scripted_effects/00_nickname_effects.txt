set_nickname_effect = {
	if = {
		limit = {
			NAND = { # I cannot already have a bad nickname if the nickname I'm getting is also bad
				has_bad_nickname = yes
				is_bad_nickname = $NICKNAME$
			}
			OR = { # Has no nickname or a bad nickname
				NOT = { has_any_nickname = yes }
				has_bad_nickname = yes
			}
		}
		give_nickname = $NICKNAME$
	}
}

#EK ADDITION: Added the epithet effects for further use, changed the name to be less confusing
ek_give_nickname_effect_character_list = {
    set_variable = {
        name = nickname_given
        value = flag:$NICKNAME$
      }
    add_to_global_variable_list = {
        name = nickname_already_given
        target = flag:$NICKNAME$
    }
}

ek_give_nickname_effect = {
	if = {
		limit = { nickname_already_given_list_trigger = { NICKNAME = $NICKNAME$ } }
		ek_give_nickname_effect_character_list = { NICKNAME = $NICKNAME$ }
		set_nickname_effect = { NICKNAME = $NICKNAME$ }
	}
}

save_culture_in_global_list = {
	$CULTURE$ = { save_scope_as = culture_to_save }
	add_to_global_variable_list = {
		name = culture_list
		target = scope:culture_to_save
	}
}

set_culture_native_to = {
	$CULTURE$ = {
		if = { # If the culture is nomadic it now isn't (nomadic divergence)
			limit = { has_innovation = innovation_native_nomadic }
			remove_innovation = innovation_native_nomadic
		}
		add_innovation = innovation_native_$REGION$
		# Special case for BM
		if = { # If we were originally from BM, we lost our special inno
			limit = { has_innovation = innovation_black_marsh_natives }
			remove_innovation = innovation_black_marsh_natives
		}
		if = { # If we just arrived in BM, we get our inno
			limit = {
				has_innovation_flag = nativity_innovation_black_marsh
				NOT = { has_innovation = innovation_black_marsh_natives }
			}
			add_innovation = innovation_black_marsh_natives
		}
	}
	add_to_global_variable_list = {
		name = culture_innovation_native_$REGION$
		target = $CULTURE$
	}
}

county_save_native_cultures = {
	every_culture_global = {
		limit = { is_native_to = { CULTURE = this REGION = $REGION$ } }
		save_temporary_scope_as = culture_to_save
		scope:current_county = {
			add_to_variable_list = {
				name = native_culture_list
				target = scope:culture_to_save
			}
		}
	}
}

ek2m_adventurer_army_maintenance = {
	effect = {
		every_independent_ruler = {
			limit = { has_government = landless_adventurer_government }
			if = {
				limit = { has_raised_armies = yes }
				remove_character_modifier = ek2m_adventurer_maa_modifier
			}
			else = {
				add_character_modifier = ek2m_adventurer_maa_modifier
			}
			
		}
	}
	on_actions = {
		delay = { months = 1 }
		ek2m_adventurer_army_maintenance
	}
}

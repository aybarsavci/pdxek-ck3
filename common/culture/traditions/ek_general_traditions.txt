tradition_sorcerer_kings = {
	category = realm

	layers = {
		0 = diplo
		1 = western
		4 = crown.dds
	}
	#Can't pick if Balfieri
	can_pick = {
		#EK TODO
		# The character needs to have skills in Arcana
		#EK NOTE - above is not implemented
		NOT = { has_cultural_tradition = tradition_direnni_sorcerer_kings }
	}
	
	parameters = {
		# Unlocks Arcanic Court Type
		unlocks_court_magical = yes
		# Unlocks Magic Succession Types
		magic_elective_enabled = yes
		# Unlocks Arcana Obligation
		vassal_contribution_tied_arcana = yes
		# Rank gives bonuses to Mages MaA
		rank_gives_bonus_to_mage_maa = yes
		# Rulers with Arcana education or congenital trait get a smol opinion bonus
		good_arcana_traits_give_bonuses = yes
		# Rulers with bad arcana congenital trait get a smol opinion malus
		bad_arcana_traits_give_maluses = yes
		# Can claim titles using Arcana
		arcana_claim_enabled = yes
		# Can Disrupt prisoners, who are then blocked from inheritance
		can_disrupt_prisoners = yes
		magic_disruption_blocks_inheritance = yes
		# Unlocks Arcane Clash tournament phase
		can_host_arcane_clash_contests = yes
		magic_preference = yes
	}
	
	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			if = {
				limit = {
					always = no
					# Not enough other rulers with some magical ability
				}
				add = {
					value = tradition_unfulfilled_criteria_penalty
					desc = king_with_vassals_desc
				}
			}
			
			if = {
				limit = {
					culture_tradition_reduction_trigger = { TRADITION = tradition_sorcerer_kings }
				}
				multiply = {
					value = 0.5
					desc = inspired_by_culture_desc
				}
			}
			
			multiply = tradition_replacement_cost_if_relevant
		}
	}
	
	ai_will_do = {
		value = 25
		if = {
			limit = {
				always = no
				# Not enough other rulers with some magical ability
			}
			multiply = 0
		}
	}
}

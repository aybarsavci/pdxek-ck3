###### BLACK MARSH ######
archonese = { #Archonese
	trigger = {
		OR = {
			AND = {
				scope:culture = { has_cultural_pillar = heritage_cyrodiilic } #Nibenese typically
				scope:other_culture = culture:horwalli
			}
			AND = {
				scope:culture = culture:horwalli
				scope:other_culture = { has_cultural_pillar = heritage_cyrodiilic }
			}
		}
	}
	hybrid = yes
}

syradri = { #Syradri - Cyrodiilic + Redguard. Taken from PT where it's used for Colo-Redguards
	trigger = {
		OR = {
			AND = {
				scope:culture = { has_cultural_pillar = heritage_yokudan }
				scope:other_culture = { has_cultural_pillar = heritage_cyrodiilic }
			}
			AND = {
				scope:culture = { has_cultural_pillar = heritage_cyrodiilic }
				scope:other_culture = { has_cultural_pillar = heritage_yokudan }
			}
		}
		capital_county = {
			title_province = { geographical_region = mundus_tamriel_cyrodiil }
		}
	}
	hybrid = yes
}

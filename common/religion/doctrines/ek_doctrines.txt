	group = "crimes"
	
	doctrine_lycanthropy_crime  = {
		piety_cost = {
			value = faith_doctrine_cost_low

				limit = { has_doctrine = doctrine_lycanthropy_crime  }

				limit = { has_doctrine = doctrine_lycanthropy_shunned }
				multiply = faith_changed_doctrine_cost_mult_two_step
			}
			else_if = {
				limit = { has_doctrine = doctrine_lycanthropy_accepted }

		}
	}
	doctrine_lycanthropy_shunned = {

			value = faith_doctrine_cost_low
			if = {
				limit = { has_doctrine = doctrine_lycanthropy_shunned }

			}
			else_if = {
				limit = { has_doctrine = doctrine_lycanthropy_virtuous }
				multiply = faith_changed_doctrine_cost_mult_two_step
			}
		}

				limit = { has_doctrine = doctrine_lycanthropy_shunned }
				multiply = faith_changed_doctrine_cost_mult_two_step
			}

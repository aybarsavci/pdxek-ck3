doctrine_azura = {
	group = "daedric_princes"

	doctrine_azura_pantheon = {
		parameters = {
			adds_azura_pantheon_active = yes
			worship_azura_allowed_active = yes
			doctrine_daedra_pantheon_crime_hostility = yes
			doctrine_daedra_pantheon_shunned_hostility = yes
		}

		can_pick = {
			trigger_if = {
				limit = { doctrine:doctrine_main_pantheon_type_monolatrism = { is_in_list = selected_doctrines } }
				creating_faith_doctrine_adds_azura = yes
			}
			trigger_else_if = {
				limit = {
					doctrine:doctrine_main_pantheon_type_polytheism = { is_in_list = selected_doctrines }
					NOT = { doctrine:doctrine_pantheon_daedra = { is_in_list = selected_doctrines } }
				}
				creating_faith_doctrine_adds_azura = yes
			}
			trigger_else = {
				always = yes
			}
		}
		
		piety_cost = {
			value = faith_doctrine_cost_high

			if = {
				limit = { has_doctrine = doctrine_azura_pantheon }
				multiply = faith_unchanged_doctrine_cost_mult
			}
			
			if = {
				limit = { has_doctrine = doctrine_azura_shunned }
				multiply = faith_changed_doctrine_cost_mult_two_step
			}
			
			if = {
				limit = { has_doctrine = doctrine_azura_criminal }
				multiply = faith_changed_doctrine_cost_mult_three_step
			}
		}
	}
	
	doctrine_azura_allowed = {
		can_pick = {
			trigger_if = {
				limit = {
					doctrine:doctrine_main_pantheon_type_daedric_prince = { is_in_list = selected_doctrines }
				}
				NOT = { doctrine:doctrine_pantheon_azura = { is_in_list = selected_doctrines } }
			}
		}
		
		parameters = {
			worship_azura_allowed_active = yes
			doctrine_daedra_allowed_crime_hostility = yes
		}
		
		piety_cost = {
			value = faith_doctrine_cost_mid

			if = {
				limit = { has_doctrine = doctrine_azura_allowed }
				multiply = faith_unchanged_doctrine_cost_mult
			}
			
			if = {
				limit = { has_doctrine = doctrine_azura_criminal }
				multiply = faith_changed_doctrine_cost_mult_two_step
			}
		}
	}
	
	doctrine_azura_shunned = {
		can_pick = {
			trigger_if = {
				limit = {
					doctrine:doctrine_main_pantheon_type_daedric_prince = { is_in_list = selected_doctrines }
				}
				NOT = { doctrine:doctrine_pantheon_azura = { is_in_list = selected_doctrines } }
			}
		}
		
		parameters = {
			worship_azura_shunned_active = yes
			doctrine_daedra_shunned_pantheon_hostility = yes
		}
		
		piety_cost = {
			value = faith_doctrine_cost_mid

			if = {
				limit = { has_doctrine = doctrine_azura_shunned }
				multiply = faith_unchanged_doctrine_cost_mult
			}
			
			if = {
				limit = { has_doctrine = doctrine_azura_pantheon }
				multiply = faith_changed_doctrine_cost_mult_two_step
			}
		}
	}
	
	doctrine_azura_criminal = {
		can_pick = {
			trigger_if = {
				limit = {
					doctrine:doctrine_main_pantheon_type_daedric_prince = { is_in_list = selected_doctrines }
				}
				NOT = { doctrine:doctrine_pantheon_azura = { is_in_list = selected_doctrines } }
			}
		}
		
		parameters = {
			worship_azura_criminal_active = yes
			doctrine_daedra_crime_pantheon_hostility = yes
			doctrine_daedra_crime_allowed_hostility = yes
		}
		
		piety_cost = {
			value = faith_doctrine_cost_mid

			if = {
				limit = { has_doctrine = doctrine_azura_criminal }

			}
			
			if = {
				limit = { has_doctrine = doctrine_azura_allowed }

			}
			
			if = {
				limit = { has_doctrine = doctrine_azura_pantheon }

			}
		}

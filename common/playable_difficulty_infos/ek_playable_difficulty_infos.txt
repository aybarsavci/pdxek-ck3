pdi_is_dead_culture = {
	is_shown = {
		culture = {
			culture_is_alive = no
			NOT = { has_innovation = innovation_daedric_race } # Dremora shouldn't get this tooltip
		}
	}
}

pdi_is_pariah_culture = {
	is_shown = {
		is_pariah_character = yes
	}
}

pdi_is_daedric_culture = {
	is_shown = {
		is_daedric_character = yes
	}
}

	is_shown = {

	}
}

# Tribal rulers have an expected level of fame. Being below it is not a good thing.
action_low_prestige_level = {
	is_dangerous = yes
	check_create_action = {
		if = {
			limit = {
				government_has_flag = government_is_tribal
				current_prestige_level < expected_prestige_level
			}
			try_create_important_action = {
				important_action_type = action_low_prestige_level
				actor = root
			}
		}
	}
}

action_autocracy_appoint_tax_collector = {
	type = alert
	icon = action_tax_collector

	check_create_action = {
		if = {
			limit = {
				any_tax_slot = {
					has_tax_collector = no
					any_tax_slot_vassal = {
						is_ruler = yes
					}
				}
				any_vassal = { government_has_flag = government_is_autocracy }
			}

			try_create_important_action = {
				important_action_type = action_autocracy_appoint_tax_collector 
				recipient = this
			}
		}
	}

	effect = {
		open_view_data = manage_tax_slots
	}
}

action_autocracy_assign_tax_payers = {
	type = alert
	icon = action_tax_collector

	check_create_action = {
		if = {
			limit = {
				employs_tax_collector = yes
				any_tax_collector = {
					count = all
					any_tax_collector_vassal = { 
						count = 0
					}
				}
			}

			try_create_important_action = {
				important_action_type = action_autocracy_assign_tax_payers
				recipient = this
			}
		}
	}

	effect = {
		open_view_data = manage_tax_slots
	}
}

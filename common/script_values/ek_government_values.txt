#Autocracy, copied from clan
autocracy_government_levies_default = 0.04
autocracy_government_levies_player = 0.25

autocracy_government_tax_default = 0.02
autocracy_government_tax_player = 0.25

autocracy_government_tax_collector_aptitude_multiplier = {
	value =	{
		if = {
			limit = {
				check_autocracy_tax_collector_aptitude = {
					CHARACTER = scope:tax_collector
					VALUE = 5
				}
			}
			add = 15
		}
		else_if = {
			limit = {
				check_autocracy_tax_collector_aptitude = {
					CHARACTER = scope:tax_collector
					VALUE = 4
				}
			}
			add = 10
		}
		else_if = {
			limit = {
				check_autocracy_tax_collector_aptitude = {
					CHARACTER = scope:tax_collector
					VALUE = 3
				}
			}
			add = 6
		}
		else_if = {
			limit = {
				check_autocracy_tax_collector_aptitude = {
					CHARACTER = scope:tax_collector
					VALUE = 2
				}
			}
			add = 3
		}
		else_if = {
			limit = {
				check_autocracy_tax_collector_aptitude = {
					CHARACTER = scope:tax_collector
					VALUE = 1
				}
			}
			add = 1
		}
		else = {
			add = 0
		}
	}
}

autocracy_government_levies_max_possible = {
	value = autocracy_government_levies_default
}

autocracy_government_tax_max_possible = {
	value = autocracy_government_tax_default
}

# Calculate the monthly change for an individuals decay level
# can make this more nuanced by making the values higher or lower based on circumstances, for example climate or stress or skill in restoration
# High arcana lowers the decay, characters with 30 or more arcana are able to slow it significantly
# The slowing tops out at 50 arcana, halving the decay rate (50 arcana x 0.0001 = 0.005, which is half of the default value)
monthly_decay_increase = {
	value = 0.01
	add = {
        value = arcana.negate
        multiply = 0.0001
    }
    min = 0.005
}

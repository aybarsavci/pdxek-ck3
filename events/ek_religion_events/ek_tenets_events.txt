##### Divine Purity
### Fired by on_birth on action: Gives the Impure trait if the child has any bad congenital trait, or if one of their parents should be considered Impure
# Gives the trait to the child, if applicable
ek_tenets.0001 = {
	type = character_event
	hidden = yes

	trigger = {
		# Must have the Divine Purity tenet
		faith = { has_doctrine_parameter = children_bad_congenital_trait_impure }

		OR = {
			# Has their body tainted
			OR = {
				has_trait = physique_bad
				has_trait = intellect_bad
				has_trait = beauty_bad
				has_trait = arcana_bad

				has_trait = clubfooted
				has_trait = hunchbacked
				has_trait = lisping
				has_trait = stuttering
				has_trait = dwarf
				has_trait = giant
				has_trait = inbred

				has_trait = spindly
				has_trait = scaly
				has_trait = albino
				has_trait = wheezing
				has_trait = bleeder
				has_trait = infertile
			}

			# Either parent is Impure / is part of a non-accepted culture (from the Cultural/Communal Identity doctrine)
			AND = {
				exists = mother
				mother = {
					OR = {
						AND = {
							root.faith = {
								OR = {
									has_doctrine = doctrine_cultural_identity_heritage
									has_doctrine = doctrine_cultural_identity_heritage_exclusive
								}
							}
							NOT = { culture = { has_same_culture_heritage = root.faith.var:associated_culture } }
						}
						AND = {
							root.faith = {
								OR = {
									has_doctrine = doctrine_cultural_identity_culture
									has_doctrine = doctrine_cultural_identity_culture_exclusive
								}
							}
							NOT = { culture = root.faith.var:associated_culture }
						}
						has_trait = impure
					}
				}
			}
			AND = {
				exists = father
				father = {
					OR = {
						AND = {
							root.faith = {
								OR = {
									has_doctrine = doctrine_cultural_identity_heritage
									has_doctrine = doctrine_cultural_identity_heritage_exclusive
								}
							}
							NOT = { culture = { has_same_culture_heritage = root.faith.var:associated_culture } }
						}
						AND = {
							root.faith = {
								OR = {
									has_doctrine = doctrine_cultural_identity_culture
									has_doctrine = doctrine_cultural_identity_culture_exclusive
								}
							}
							NOT = { culture = root.faith.var:associated_culture }
						}
						has_trait = impure
					}
				}
			}
		}
	}

	immediate = {
		# Give the trait
		add_trait = impure
		save_scope_as = impure_character

		# Notifies the parents
		if = {
			limit = { exists = mother }
			mother = { trigger_event = ek_tenets.0002 }
		}

		if = {
			limit = { exists = father }
			father = { trigger_event = ek_tenets.0002 }
		}
	}
}

### Exalted Heroism
## Character with max level of Fame and Devotion becomes a new Saint
ek_tenets.0051 = {
	type = character_event
	content_source = dlc_ek
	title = ek_tenets.0051.t
	desc = ek_tenets.0051.desc

	theme = faith

	left_portrait = {
		character = root
	}

	immediate = {
		add_character_flag = had_ek_tenets.0051 #prevent it from triggering several times
	}
	
	trigger = {
		# Your faith must allow new Saints
		faith = { has_doctrine_parameter = new_saints }
		piety_level >= max_piety_level
		prestige_level >= max_prestige_level
		NOT = { has_trait = saint }
		NOT = { has_character_flag = had_ek_tenets.0051 }
	}

	option = {
		name = ek_tenets.0051.a

		char_becomes_saint = yes
	}
}

## New Saints can choose a new virtue or sin
ek_tenets.0052 = {
	type = character_event
	content_source = dlc_ek
	title = ek_tenets.0052.t
	desc = ek_tenets.0052.desc

	theme = faith

	left_portrait = {
		character = root
	}

	trigger = {
		# Your faith must allow new Saints
		faith = { has_doctrine_parameter = virtues_sins_decided_saints }
		has_trait = saint
		NOT = { has_character_flag = already_changed_virtues_sins_saint }
	}

	immediate = {
		add_character_flag = { flag = already_changed_virtues_sins_saint } # So the event doesn't repeat
	}

	### NOTE: Each faith can have up to three virtues or sins, so by adding a new sin or virtue you need to replace an already existing one
	# A new virtue
	option = {
		name = ek_tenets.0052.a

		custom_tooltip = ek_tenets.0052.swapping_virtue

		trigger_event = ek_tenets.0053 # Which virtue to replace?
	}
	# A new sin
	option = {
		name = ek_tenets.0052.b

		custom_tooltip = ek_tenets.0052.swapping_sin

		trigger_event = ek_tenets.0055 # Which sin to replace?
	}
}

#Converting to Sixth House
ek_tenets.0060 = {
	type = character_event
	title = ek_tenets.0060.t
	desc = ek_tenets.0060.desc
	theme = faith
	override_background = corridor_night

	right_portrait = {
		character = root
		animation = personality_zealous
	}

	immediate = { #Remove secret sixth house and make it open
		if = {
			limit = { 
				secret_faith ?= faith:sixth_house
			}
			random_secret = {
				limit = { secret_type = secret_crypto_religionist }
				remove_secret = yes
			}
			set_character_faith = faith:sixth_house
		}
	}

	option = {
		name = ek_tenets.0060.a
		add_piety = 500
		add_prestige = 500
	}
}

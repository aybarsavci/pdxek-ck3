namespace = ek_holy_orders

ek_holy_orders.1 = {
	scope = none
	hidden = yes
	
	trigger = {
		game_start_date >= 3360.1.1
	}	
	
	immediate = {
		title:c_bethrimo = { save_scope_as = county }
		if = {
			limit = { exists = title:e_tribunal_temple.holder }
			title:e_tribunal_temple.holder = { save_temporary_scope_as = sponsor }
		}
		else = {
			scope:county.holder = { save_temporary_scope_as = sponsor }
		}
		
		if = {
			limit = { 
				exists = character:indoril_99b1
				character:indoril_99b1 =  { is_alive = yes } 
			}
			character:indoril_99b1 = { save_scope_as = leader }
		}
		if = {
			limit = {
				current_date >= 3441.1.1
				exists = character:sedroth1_991
				character:sedroth1_991 =  { is_alive = yes } 
			}
			character:sedroth1_991 = { save_scope_as = leader }
		}
		
		scope:leader = {
			create_historical_holy_order_effect = { SPONSOR = scope:sponsor COUNTY = scope:county ORDER = d_ordinator_order_of_war }
			historical_lease_out_effect = { COUNTY = title:c_drakehold ORDER = scope:new_holy_order }
		}
	}
}

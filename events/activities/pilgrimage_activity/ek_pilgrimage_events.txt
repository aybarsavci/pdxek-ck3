# Sermon Reading
ek_pilgrimage.0001 = {
	type = character_event
	content_source = dlc_ek
	title = ek_pilgrimage.0001.t
	desc = ek_pilgrimage.0001.desc
	theme = travel_pilgrimage
	override_background = { reference = terrain_travel }
	left_portrait = {
		character = root
		triggered_animation = {
			trigger = { has_trait = cynical }
			animation = personality_cynical
		}
		triggered_animation = {
			trigger = { has_trait = zealous }
			animation = admiration
		}
		animation = personality_honorable
	}

	right_portrait = {
		character = scope:sermon_reader
		animation = personality_zealous
		outfit_tags = { bishop_outfit }
	}
	cooldown = { years = yearly_cooldown }
	trigger = {
		faith = {
			has_doctrine = doctrine_pantheon_tribunal
		}
		is_location_valid_for_travel_event_on_land = yes
		NOT = {
			exists = scope:activity.var:had_ek_pilgrimage_0001
		}
	}

	immediate = {
		random_dummy_gender_clergy_effect = yes
		create_character = {
			age = {25 55}
			gender = scope:dummy_clergy_gender
			location = root.location
			faith = root.faith
			culture = root.culture
			#EK EDIT
			after_creation = {
				ek_character_setup_effect = yes
			}
			#EK EDIT END
			save_scope_as = sermon_reader
		}
		scope:activity = {
			set_variable = {
				name = had_ek_pilgrimage_0001
				value = yes
			}
		}
	}

	option = {
		name = ek_pilgrimage.0001.a
		add_piety = minor_piety_gain
		give_pilgrim_xp_effect = yes
		stress_impact = {
			cynical = medium_stress_impact_gain
		}			
	}

	option = {
		name = ek_pilgrimage.0001.b

		add_piety = miniscule_piety_loss
		stress_impact = {
			zealous = medium_stress_impact_gain
			cynical = medium_stress_impact_loss
		}
	}

	after = {
		hidden_effect = {
			scope:sermon_reader = {
				if = {
					limit = {
						root = { is_ai = yes }
					}
					silent_disappearance_effect = yes
				}
			}
		}
	}
}

				name = had_ek_pilgrimage_0003

			}
		}
		add_character_flag = in_pilgrimage_event_chain
	}

	option = {
		name = ek_pilgrimage.0003.a

			cynical = major_stress_impact_gain

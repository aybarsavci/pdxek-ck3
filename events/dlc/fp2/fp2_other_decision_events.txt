			limit = {
				exists = var:holding_quality
				var:holding_quality > 1
			}

			limit = {
				exists = var:holding_quality
				var:holding_quality > 2
			}
			every_ruler = {
				limit = {
					in_diplomatic_range = scope:host
				}

					faith = {
						if = {
							limit = {
								NOT = {
									any_in_list = {
										list = potential_faiths
									}
								}
							}
							add_to_list = potential_faiths

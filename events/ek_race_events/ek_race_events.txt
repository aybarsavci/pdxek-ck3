namespace = ek_race

#If we ever need something to be applied first or to delay the other events
ek_race.0002 = {
	scope = none
	hidden = yes

	immediate = {
		every_living_character = {
			trigger_event = { id = ek_race.0003 }
		}
	}
}

#Directs newly-generated characters to the setup effect (race, lifespan and birthsign)
ek_race.0003 = {
	hidden = yes
	
	trigger = { NOT = { has_character_flag = char_setup } }

	immediate = {
		hidden_effect = { ek_character_setup_effect = yes }
	}
}

#Directs newborns directly to the race & lifespan inheritance effect
ek_race.0004 = {
	hidden = yes

	trigger = { NOT = { has_character_flag = char_setup } }

	immediate = {
		hidden_effect = {
			race_traits_inheritance_effect = yes
			lifespan_traits_inheritance_effect = yes
		}
	}
}

#Pregnancy race compatibility check
ek_race.0100 = {
	hidden = yes

	immediate = {
	
		if = {
			#Race check
			limit = { 
				AND = { 
					ek_race_compatible_trigger = { CHARACTER1 = scope:mother CHARACTER2 = scope:real_father } 
					NOT = { has_character_flag = race_daedra }
				} 
			}
			
			#Vanilla pregnancy on_action, the specific triggers are in each event
			trigger_event = pregnancy.0001 # Universal hidden maintenance event 
			trigger_event = pregnancy.1001 # Mother gives birth to current husband's child. 
			trigger_event = pregnancy.1101 # Mother gives birth to ex-husband's child. 
			trigger_event = pregnancy.2001 # Unwed mother gives birth to bastard child. 
			trigger_event = pregnancy.2002 # Married mother gives birth to bastard child.
			trigger_event = pregnancy.2050 # Married mother in same-sex relationship.
		}
		else = { 
			end_pregnancy = yes
			pregnancy_cleanup_effect = yes
		}
	}
}

#Checks the need to kill leftover incompatible babies at birth
ek_race.0101 = {
	hidden = yes

	immediate = {
		if = { 
			#Race check
			limit = { 
				AND = { 
					ek_race_compatible_trigger = { CHARACTER1 = scope:mother CHARACTER2 = scope:real_father } 
					NOT = { has_character_flag = race_daedra }
				} 
			}
			
			#Vanilla birth on_action, the specific triggers are in each event
			trigger_event = birth.0001
			trigger_event = birth.3011
		}
		else = { trigger_event = ek_race.0102 }
	}
}

#Kills any leftover incompatible babies at birth
ek_race.0102 = {
	title = birth.3001.t
	desc = birth.3001.desc_other_child
	theme = death
	
	left_portrait = {
		character = root
		animation = sadness
	}
	right_portrait = {
		character = scope:father
		trigger = {
			exists = scope:father
		}
		animation = disbelief
	}
	
	immediate = {
		play_music_cue = mx_cue_death
		hidden_effect = {
			scope:child = { death = { death_reason = death_stillborn } }
		}
		if = {
			limit = {
				exists = scope:father
				scope:father = { is_alive = yes }
			}
			scope:father = { trigger_event = birth.3002 }
		}
		custom_tooltip = birth.3001.a.tt
	}

	option = {
		name = birth.3001.a
	}
}

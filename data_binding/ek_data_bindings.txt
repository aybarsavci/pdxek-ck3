#For Arcane Clash Tournament GUI
macro = { 
	description = "Is the datacontexted contest a magic contest?"
	definition = "IsMagicContest"
	replace_with = "IsContestType( 'tournament_phase_arcane_clash' )"
}

macro = {
	description = "Gets the script value of the datacontexted character's aptitude for the specified contest"
	definition = "GetMagicContestTypeAptitude"
	replace_with = "GetContestAptitude( 'contest_type_aptitude_magic' )"
}

#EK ADDITIONS: new court rooms and edited vanilla court triggers

######################
### EK court rooms ###
######################

white_gold = {	# location specific court room
	trigger = {
		exists = capital_county
		capital_county = title:c_imperial_city
	}
}

yespest = {
	trigger = {
		NOT = { capital_county ?= title:c_imperial_city }
		capital_county.culture = {
			OR = {
				has_graphical_ayleid_culture_group_trigger = yes
			}
		}
	}
}

nord = {
	trigger = {
		NOT = { capital_county ?= title:c_imperial_city }
		capital_county.culture = {
			has_graphical_nordic_culture_group_trigger = yes
		}
	}
}

mediterranean = {
	trigger = {
		NOT = { capital_county ?= title:c_imperial_city }

				has_graphical_cyrodilic_culture_group_trigger = yes

			}
		}
	}
